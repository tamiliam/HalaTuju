# Contract Module — Sprint 5 prod cutover SQL (migration 0103, migrate-first)

Applied to HalaTuju prod (`pbrrlyoyyiftckqvzvvo`, Postgres 17) via the Supabase MCP
`apply_migration`, migrate-first (before the code deploy). DDL grounded in the prod
conventions of the existing Django-5.2 tables (`bigint … IDENTITY` PKs, `jsonb`,
`timestamptz`, FKs `DEFERRABLE INITIALLY DEFERRED`, RLS enabled with no policies =
deny-by-default for the PostgREST roles; the Django service role bypasses). Exact
`UniqueConstraint` names from the migration. `django_migrations` row recorded in the
same transaction. Verified after apply by running `seed_contract_template` on prod
(the ORM exercises every column) — draft only, no PII.

```sql
-- bursary_agreements: three additive columns (table has 0 rows on prod)
ALTER TABLE bursary_agreements ADD COLUMN drive_file_url varchar(200) NOT NULL DEFAULT '';
ALTER TABLE bursary_agreements ALTER COLUMN drive_file_url DROP DEFAULT;
ALTER TABLE bursary_agreements ADD COLUMN executed_pdf_emailed_at timestamptz NULL;

-- contract_templates
CREATE TABLE contract_templates (
    id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    version varchar(40) NOT NULL,
    status varchar(20) NOT NULL,
    title_en varchar(255) NOT NULL,
    title_ms varchar(255) NOT NULL,
    title_ta varchar(255) NOT NULL,
    preamble_en text NOT NULL,
    preamble_ms text NOT NULL,
    preamble_ta text NOT NULL,
    progress_standard_en text NOT NULL,
    progress_standard_ms text NOT NULL,
    progress_standard_ta text NOT NULL,
    counterparty_name varchar(200) NOT NULL,
    counterparty_title varchar(255) NOT NULL,
    counterparty_nric varchar(20) NOT NULL,
    counterparty_notify_emails jsonb NOT NULL,
    parent_role varchar(20) NOT NULL,
    parent_pin_required boolean NOT NULL,
    witness_policy varchar(10) NOT NULL,
    vetted_by_name varchar(200) NOT NULL,
    vetted_on date NULL,
    vetting_attested_by_email varchar(254) NOT NULL,
    vetting_attested_at timestamptz NULL,
    created_by_email varchar(254) NOT NULL,
    submitted_by_email varchar(254) NOT NULL,
    submitted_by_at timestamptz NULL,
    deployed_by_email varchar(254) NOT NULL,
    deployed_by_at timestamptz NULL,
    archived_at timestamptz NULL,
    created_at timestamptz NOT NULL,
    updated_at timestamptz NOT NULL,
    organisation_id bigint NOT NULL,
    CONSTRAINT uniq_contract_template_org_version UNIQUE (organisation_id, version),
    CONSTRAINT contract_templates_organisation_id_fkey
        FOREIGN KEY (organisation_id) REFERENCES partner_organisations(id) DEFERRABLE INITIALLY DEFERRED
);

-- contract_clauses
CREATE TABLE contract_clauses (
    id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    "order" integer NOT NULL CONSTRAINT contract_clauses_order_check CHECK ("order" >= 0),
    heading_en varchar(255) NOT NULL,
    heading_ms varchar(255) NOT NULL,
    heading_ta varchar(255) NOT NULL,
    body_en text NOT NULL,
    body_ms text NOT NULL,
    body_ta text NOT NULL,
    is_quiz_candidate boolean NOT NULL,
    quiz_en jsonb NOT NULL,
    quiz_ms jsonb NOT NULL,
    quiz_ta jsonb NOT NULL,
    quiz_generated_model varchar(80) NOT NULL,
    template_id bigint NOT NULL,
    CONSTRAINT uniq_contract_clause_template_order UNIQUE (template_id, "order"),
    CONSTRAINT contract_clauses_template_id_fkey
        FOREIGN KEY (template_id) REFERENCES contract_templates(id) DEFERRABLE INITIALLY DEFERRED
);

-- contract_payment_schedule_rows
CREATE TABLE contract_payment_schedule_rows (
    id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    pathway varchar(40) NOT NULL,
    variant varchar(20) NOT NULL,
    label_en varchar(120) NOT NULL,
    label_ms varchar(120) NOT NULL,
    label_ta varchar(120) NOT NULL,
    monthly_amount numeric(10,2) NOT NULL,
    start_month smallint NOT NULL CONSTRAINT contract_sched_start_month_check CHECK (start_month >= 0),
    paid_offsets jsonb NOT NULL,
    sort_order integer NOT NULL CONSTRAINT contract_sched_sort_order_check CHECK (sort_order >= 0),
    template_id bigint NOT NULL,
    CONSTRAINT uniq_contract_schedule_template_pathway_variant UNIQUE (template_id, pathway, variant),
    CONSTRAINT contract_sched_template_id_fkey
        FOREIGN KEY (template_id) REFERENCES contract_templates(id) DEFERRABLE INITIALLY DEFERRED
);

-- bursary_agreements.template_id + scholarship_applications.comprehension_template_id (nullable FKs)
ALTER TABLE bursary_agreements ADD COLUMN template_id bigint NULL;
ALTER TABLE bursary_agreements ADD CONSTRAINT bursary_agreements_template_id_fkey
    FOREIGN KEY (template_id) REFERENCES contract_templates(id) DEFERRABLE INITIALLY DEFERRED;
CREATE INDEX bursary_agreements_template_id_idx ON bursary_agreements (template_id);

ALTER TABLE scholarship_applications ADD COLUMN comprehension_template_id bigint NULL;
ALTER TABLE scholarship_applications ADD CONSTRAINT scholarship_apps_comprehension_template_id_fkey
    FOREIGN KEY (comprehension_template_id) REFERENCES contract_templates(id) DEFERRABLE INITIALLY DEFERRED;
CREATE INDEX scholarship_apps_comprehension_template_id_idx ON scholarship_applications (comprehension_template_id);

-- RLS: enable on the three new tables, no policies (deny-by-default for PostgREST roles;
-- the Django service-role connection bypasses RLS). Mirrors payment_runs.
ALTER TABLE contract_templates ENABLE ROW LEVEL SECURITY;
ALTER TABLE contract_clauses ENABLE ROW LEVEL SECURITY;
ALTER TABLE contract_payment_schedule_rows ENABLE ROW LEVEL SECURITY;

-- Record the Django migration (migrate-first convention).
INSERT INTO django_migrations (app, name, applied)
VALUES ('scholarship', '0103_contract_module', now());
```

**Note:** the FK columns that are the FIRST column of a unique constraint
(`organisation_id`, `template_id` in clauses/schedule) get their index from that unique
constraint — no separate FK index needed (matches Django). Only the two AddField FKs
(`bursary_agreements.template_id`, `scholarship_applications.comprehension_template_id`)
need an explicit index.
